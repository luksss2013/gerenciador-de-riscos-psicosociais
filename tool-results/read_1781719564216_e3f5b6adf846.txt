    98тЖТ    98тЖТfunction deriveStatus(c: CompanySummary): NrStatus {
    99тЖТ    99тЖТ  const { lastAssessmentStatus, lastAssessmentCompletedAt } = c.summary;
   100тЖТ   100тЖТ  if (!lastAssessmentStatus) return "no_assessment";
   101тЖТ   101тЖТ  if (lastAssessmentStatus === "collecting") return "collecting";
   102тЖТ   102тЖТ  if (lastAssessmentStatus === "processing") return "processing";
   103тЖТ   103тЖТ  if (lastAssessmentStatus === "draft") return "draft";
   104тЖТ   104тЖТ  if (lastAssessmentStatus === "archived") return "archived";
   105тЖТ   105тЖТ  if (lastAssessmentStatus === "completed") {
   106тЖТ   106тЖТ    if (lastAssessmentCompletedAt) {
   107тЖТ   107тЖТ      const age = Date.now() - new Date(lastAssessmentCompletedAt).getTime();
   108тЖТ   108тЖТ      if (age > TWO_YEARS_MS) return "review_recommended";
   109тЖТ   109тЖТ    }
   110тЖТ   110тЖТ    return "completed";
   111тЖТ   111тЖТ  }
   112тЖТ   112тЖТ  return "no_assessment";
   113тЖТ   113тЖТ}

   115тЖТ   115тЖТfunction fmtDate(iso: string | null, fallback = "тАФ"): string {
   116тЖТ   116тЖТ  if (!iso) return fallback;
   117тЖТ   117тЖТ  try {
   118тЖТ   118тЖТ    const d = parseISO(iso);
   119тЖТ   119тЖТ    if (!isValidDate(d)) return fallback;
   120тЖТ   120тЖТ    return format(d, "dd/MM/yyyy", { locale: ptBR });
   121тЖТ   121тЖТ  } catch {
   122тЖТ   122тЖТ    return fallback;
   123тЖТ   123тЖТ  }
   124тЖТ   124тЖТ}

   126тЖТ   126тЖТfunction fmtPeriod(start: string | null, end: string | null): string {
   127тЖТ   127тЖТ  const s = start ? fmtDate(start) : "тАФ";
   128тЖТ   128тЖТ  const e = end ? fmtDate(end) : "тАФ";
   129тЖТ   129тЖТ  return `${s} тЖТ ${e}`;
   130тЖТ   130тЖТ}
